# 1. Простое действие

Minimal Action:

```python
from dataclasses import dataclass
from ActionMachine.Core.BaseAction import BaseAction
from ActionMachine.Core.BaseParams import BaseParams
from ActionMachine.Core.BaseResult import BaseResult
from ActionMachine.Core.AspectMethod import aspect, summary_aspect

class DoubleNumber(BaseAction):

    @dataclass(frozen=True)
    class Params(BaseParams):
        value: int

    @dataclass(frozen=True)
    class Result(BaseResult):
        doubled: int

    @summary_aspect("Удвоение числа")
    def handle(self, params, state, deps):
        return DoubleNumber.Result(params.value * 2)
```

✔ Все базовые классы имеют рейтинг A по radon MI [1].

# 2. Действие с несколькими аспектами

```python
class CreateOrder(BaseAction):

    @dataclass(frozen=True)
    class Params(BaseParams):
        user_id: str
        product_id: int

    @dataclass(frozen=True)
    class Result(BaseResult):
        order_id: str

    @aspect("Проверка ID товара")
    def validate_product(self, params, state, deps):
        if params.product_id <= 0:
            raise ValueError("Invalid product_id")
        state["valid"] = True
        return state

    @summary_aspect("Создание")
    def handle(self, params, state, deps):
        return CreateOrder.Result(order_id=f"order-{params.user_id}-{params.product_id}")
```

✔ Обработка аспектов машина выполняет через `_execute_regular_aspects`, рейтинг B (ожидаемая сложность orchestration-кода) [1].

# 3. Действие с зависимостями (@depends)

```python
from ActionMachine.Core.AspectMethod import depends

class EmailService:
    def send(self, to, msg):
        print(to, msg)

@depends(EmailService, description="Email sender")
class Notify(BaseAction):

    @dataclass(frozen=True)
    class Params(BaseParams):
        email: str
        message: str

    @dataclass(frozen=True)
    class Result(BaseResult):
        ok: bool

    @summary_aspect("Уведомление")
    def handle(self, params, state, deps):
        email_service = deps.get(EmailService)
        email_service.send(params.email, params.message)
        return Notify.Result(ok=True)
```

✔ DependencyFactory имеет рейтинг A (71.46 MI) — очень высокий показатель качества [1].

# 4. Вложенные действия через deps.run_action

```python
@depends(DoubleNumber)
class UseChild(BaseAction):

    @dataclass(frozen=True)
    class Params(BaseParams):
        x: int

    @dataclass(frozen=True)
    class Result(BaseResult):
        final: int

    @summary_aspect("Используем DoubleNumber")
    def handle(self, params, state, deps):
        child_res = deps.run_action(DoubleNumber, DoubleNumber.Params(params.x))
        return UseChild.Result(final=child_res.doubled + 5)
```

✔ Вложенность корректно отображается плагинами с учётом nest_level [1].

# 5. Действие с ролями

```python
from ActionMachine.Auth.CheckRoles import CheckRoles

@CheckRoles(CheckRoles.ANY, desc="Любой авторизованный пользователь")
class SecureAction(BaseAction):

    @dataclass(frozen=True)
    class Params(BaseParams):
        value: int

    @dataclass(frozen=True)
    class Result(BaseResult):
        res: int

    @summary_aspect("Доступный метод")
    def handle(self, params, state, deps):
        return SecureAction.Result(params.value * 10)
```

✔ Все проверки `_check_any_role`, `_check_single_role`, `_check_list_role` имеют рейтинг A (1–3) [1].

# 6. Чекеры (валидация)

```python
from ActionMachine.Checkers.StringFieldChecker import StringFieldChecker
from ActionMachine.Checkers.IntFieldChecker import IntFieldChecker

class Register(BaseAction):

    @dataclass(frozen=True)
    @StringFieldChecker("username", desc="Имя пользователя", not_empty=True)
    @IntFieldChecker("age", desc="Возраст", min_value=1)
    class Params(BaseParams):
        username: str
        age: int

    @dataclass(frozen=True)
    class Result(BaseResult):
        ok: bool

    @summary_aspect("Регистрация")
    def handle(self, params, state, deps):
        return Register.Result(ok=True)
```

✔ Все чекеры String/Int/Bool/Date получили рейтинг A в radon [1].

# 7. Плагины — логирование, метрики, мониторинг

```python
from ActionMachine.Plugins.Plugin import Plugin
from ActionMachine.Plugins.Decorators import on

class LogPlugin(Plugin):
    def get_initial_state(self):
        return {}

    @on("global_start", ".*", ignore_exceptions=True)
    async def on_start(self, state, event, action_name, params, *_):
        print("[start]", action_name)
        return state

    @on("global_finish", ".*", ignore_exceptions=True)
    async def on_finish(self, state, event, action_name, params, state_aspect, is_summary, deps, context, result, duration, nest, **_):
        print("[finish]", action_name, "duration:", duration)
        return state
```

✔ Плагины имеют рейтинг A (Plugin — 59.09 MI, Decorators — 71.41 MI) [1].  
✔ Поддерживают вложенность, асинхронность, ignore_exceptions.

# 8. Запуск через ActionProductMachine

```python
from ActionMachine.Core.ActionProductMachine import ActionProductMachine
from ActionMachine.Context.Context import Context

machine = ActionProductMachine(context=Context(), plugins=[LogPlugin()])
action = DoubleNumber()

result = machine.run(action, DoubleNumber.Params(21))
print(result.doubled)  # 42
```

✔ ActionProductMachine общая оценка: A (50.65 MI) [1].

# 9. Тестирование через ActionTestMachine

```python
from ActionMachine.Core.ActionTestMachine import ActionTestMachine

machine = ActionTestMachine({
    DoubleNumber: DoubleNumber.Result(999)
})

parent = UseChild()
params = UseChild.Params(10)

result = machine.run(parent, params)
assert result.final == 999 + 5
```

✔ Файл ActionTestMachine имеет оценку A (72.83 MI) — идеален для тестов [1].
