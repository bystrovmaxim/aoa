```python
from abc import ABC, abstractmethod
from typing import Any

class BaseLogger(ABC):

    @abstractmethod
    def warning(self, message: str, **context: Any) -> None: ...
    
    @abstractmethod
    def error(self, message: str, **context: Any) -> None: ...
```

### Координатор — единственная точка входа

```python
# ActionMachine/Logging/LogCoordinator.py

from typing import Any, List, Optional
from .BaseLogger import BaseLogger
from .PrintLogger import PrintLogger

class LogCoordinator:
    """
    Единственная точка логирования в AOA.
    В тестах — PrintLogger.
    В бою — любые логеры через add_logger().
    """
    
    def __init__(self) -> None:
        self._loggers: List[BaseLogger] = []
    
    def add_logger(self, logger: BaseLogger) -> None:
        self._loggers.append(logger)
    
    def info(self, message: str, **context: Any) -> None:
        for logger in self._loggers:
            logger.info(message, **context)
    
    def warning(self, message: str, **context: Any) -> None:
        for logger in self._loggers:
            logger.warning(message, **context)
    
    def error(self, message: str, **context: Any) -> None:
        for logger in self._loggers:
            logger.error(message, **context)
    
    def debug(self, message: str, **context: Any) -> None:
        for logger in self._loggers:
            logger.debug(message, **context)


# Глобальный экземпляр — как в logging.getLogger()
_coordinator: Optional[LogCoordinator] = None

def get_logger() -> LogCoordinator:
    global _coordinator
    if _coordinator is None:
        # По умолчанию — PrintLogger (удобно для разработки)
        _coordinator = LogCoordinator()
        _coordinator.add_logger(PrintLogger())
    return _coordinator

def configure_logger(coordinator: LogCoordinator) -> None:
    """Заменяет глобальный координатор."""
    global _coordinator
    _coordinator = coordinator
```

## PostgresUnitOfWork — для kanban_assistant

Тебе нужна простая буферизация команд:

```python
# ActionMachine/ResourceManagers/PostgresUnitOfWork.py

from typing import Any, Dict, List, Optional, Tuple, Type
import asyncpg
from ActionMachine.Core.Exceptions import HandleException
from .IConnectionManager import IConnectionManager
from .WrapperConnectionManager import WrapperConnectionManager

class PostgresUnitOfWork(IConnectionManager):
    """
    Менеджер соединения с буферизацией команд.
    execute() накапливает команды в буфер.
    commit() отправляет всё одним batch.
    select() делает flush перед чтением.
    """
    
    def __init__(self, connection_params: Dict[str, Any]) -> None:
        self._params = connection_params
        self._conn: Optional[asyncpg.Connection] = None
        self._buffer: List[Tuple[str, Optional[Tuple[Any, ...]]]] = []
        self._batch_size: int = 100
    
    async def open(self) -> None:
        try:
            self._conn = await asyncpg.connect(**self._params)
            await self._conn.execute("BEGIN")
        except Exception as e:
            raise HandleException(f"Ошибка подключения: {e}") from e
    
    async def execute(
        self,
        query: str,
        params: Optional[Tuple[Any, ...]] = None
    ) -> None:
        """Добавляет команду в буфер. В БД не отправляет."""
        self._buffer.append((query, params))
        
        # Авто-flush при достижении лимита
        if len(self._buffer) >= self._batch_size:
            await self._flush()
    
    async def select(
        self,
        query: str,
        params: Optional[Tuple[Any, ...]] = None
    ) -> List[Any]:
        """Flush буфера перед SELECT — данные всегда актуальны."""
        await self._flush()
        
        if self._conn is None:
            raise HandleException("Соединение не открыто")
        try:
            return await self._conn.fetch(query, *(params or ()))
        except Exception as e:
            raise HandleException(f"Ошибка SELECT: {e}") from e
    
    async def _flush(self) -> None:
        """Отправляет буфер в БД."""
        if not self._buffer or self._conn is None:
            return
        try:
            for query, params in self._buffer:
                await self._conn.execute(query, *(params or ()))
            self._buffer.clear()
        except Exception as e:
            raise HandleException(f"Ошибка flush: {e}") from e
    
    async def commit(self) -> None:
        """Flush + COMMIT."""
        await self._flush()
        if self._conn is None:
            raise HandleException("Соединение не открыто")
        try:
            await self._conn.execute("COMMIT")
        except Exception as e:
            raise HandleException(f"Ошибка commit: {e}") from e
        finally:
            await self._conn.close()
            self._conn = None
    
    async def rollback(self) -> None:
        self._buffer.clear()  # буфер тоже сбрасываем
        if self._conn is None:
            return
        try:
            await self._conn.execute("ROLLBACK")
        except Exception as e:
            raise HandleException(f"Ошибка rollback: {e}") from e
        finally:
            await self._conn.close()
            self._conn = None
    
    def get_wrapper_class(self) -> Optional[Type[IConnectionManager]]:
        return WrapperConnectionManager
```

**Вариант через DI** — логер не знает контекст выполнения:

Причём это **автоматически** — разработчик пишет только сообщение [1].

## Как это реализуется в `ActionProductMachine`

`ActionMachine` создаёт `ContextualLogger` перед каждым вызовом аспекта и **инжектирует его как шестой аргумент**:

```python
async def _call_aspect(
    self,
    method: AspectMethod,
    action: BaseAction,
    params: BaseParams,
    state: Dict,
    factory: DependencyFactory,
    connections: Dict,
) -> Any:
    # Создаём контекстуальный логер для этого аспекта
    log = ContextualLogger(
        coordinator=self._log_coordinator,
        nest_level=self._nest_level,
        action_name=action.get_full_class_name(),
        aspect_name=method.__name__,
    )
    
    return await method(action, params, state, factory, connections, log)
```

## ContextualLogger — что он знает

```python
class ContextualLogger:
    """
    Логер с полным знанием контекста выполнения.
    Создаётся машиной автоматически перед каждым аспектом.
    """
    
    def __init__(
        self,
        coordinator: LogCoordinator,
        nest_level: int,
        action_name: str,
        aspect_name: str,
    ) -> None:
        self._coordinator = coordinator
        self._nest_level = nest_level
        self._action_name = action_name
        self._aspect_name = aspect_name
    
    def info(self, message: str, **context: Any) -> None:
        self._coordinator.info(
            message,
            nest_level=self._nest_level,
            action=self._action_name,
            aspect=self._aspect_name,
            **context
        )
    
    def warning(self, message: str, **context: Any) -> None:
        self._coordinator.warning(message, **self._enrich(context))
    
    def error(self, message: str, **context: Any) -> None:
        self._coordinator.error(message, **self._enrich(context))
    
    def debug(self, message: str, **context: Any) -> None:
        self._coordinator.debug(message, **self._enrich(context))
    
    def _enrich(self, context: Dict) -> Dict:
        return {
            "nest_level": self._nest_level,
            "action": self._action_name,
            "aspect": self._aspect_name,
            **context
        }
```

## Почему это элегантно

Посмотри на то что ты придумал в контексте всей архитектуры [1]:

```
Params    — что приходит в Action
State     — что накапливается внутри
```

**Пять параметров — полная картина мира аспекта.**

Причём `log` — единственный из пяти кто знает контекст **самой машины**:

- `action_name` — в каком Action
- `aspect_name` — в каком шаге
