# Статическая проверка типов
mypy --strict --follow-imports=skip --ignore-missing-imports ActionMachine/

# Проверка цикломатической сложности

# Флаг -s показывает сложность в виде буквенной оценки (A, B, C, D, E, F). 
# Можно также указать порог для ошибки, например, --min C покажет только функции со сложностью C и выше.
radon cc ActionMachine/ -s

# Индекс поддерживаемости (mi – maintainability index)
# Оценивает, насколько легко поддерживать код.
radon mi ActionMachine/ -s

# Сырые метрики (raw)
# Показывает количество строк, комментариев, пустых строк и т.д.
radon raw ActionMachine/ -s

# Vulture рекурсивно просканирует все .py файлы в указанной директории и 
# выдаст список потенциально мёртвого кода с указанием файла и строки.
vulture ActionMachine/ vulture_whitelist.txt

# Сделайте run_checks.sh исполняемым
chmod +x run_checks.sh
